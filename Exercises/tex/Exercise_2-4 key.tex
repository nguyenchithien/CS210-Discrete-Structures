        \textbf{Step 2:}
        
            $s_{n} = s_{n-1} + 3n^{2}$
            
        \textbf{Solution:}
        
            $s_{1} = $ {3}{}
            \tab[4cm]

    \item
        \textbf{Step 1:}
            $s_{1} = \sum_{i=1}^{1} (2^{i-1}+1) = 2^{0} + 1 = 2$
        
        \textbf{Step 2:}
            $s_{n} = s_{n-1} + 2^{n-1} + 1$
        
        \textbf{Solution:}
                $s_{1} = $ {2}{}
                \tab[4cm]
                $s_{n} = $ {$s_{n-1} + 2^{n-1} + 1$}{}

    \item
        \textbf{Step 1:}
                $D(m) = 2m$
        
        \textbf{Step 2:}
                $D(1) = 2(1) = 2$ \checkmark
        
        \textbf{Step 3:}
                $D(m-1) = 2(m-1) = 2m - 2$
        
        \textbf{Step 4:}
                $D(m-1) = 2m - 2$
        
        \textbf{Step 5:}
                $D(m-1) = D(m) - 2$
        
        \textbf{Step 6:}
                $D(m) = D(m-1) + 2$
        
        \textbf{Step 7:}
                $D(m) = 2k + 2$
        
        \textbf{Step 8:}
                $D(m) = 2(k + 1)$

    \item
        \textbf{Step 1:}
                $D(m) = 4m+1$
        
        \textbf{Step 2:}
                $D(1) = 4(1) + 1 = 5$ \checkmark
        
        \textbf{Step 3:}
                $D(m-1) = 4(m-1) + 1 = 4m - 4 + 1$
        
        \textbf{Step 4:}
                $D(m-1) = (4m + 1) - 4$
        
        \textbf{Step 5:}
                $D(m-1) = D(m) - 4$
        
        \textbf{Step 6:}
                $D(m) = D(m-1) + 4$
        
        \textbf{Step 7:}
                $D(m) = (2k+1) + 4 = 2k + 4 + 1$
        
        \textbf{Step 8:}
                $D(m) = 2(k + 2) + 1$

    \item
        \textbf{Step 1:}
                $D(m) = m^{2} - m$
        
        \textbf{Step 2:}
                $D(1) = 1^{2} - 1 = 0$ \checkmark
        
        \textbf{Step 3:}
                $D(m-1) = (m-1)^{2} - (m-1) = m^{2} - 2m + 1 - m + 1$
        
        \textbf{Step 4:}
                $D(m-1) = (m^{2} - m) - 2m + 2$
        
        \textbf{Step 5:}
                $D(m-1) = D(m) - 2m  + 2$
        
        \textbf{Step 6:}
                $D(m) = D(m-1) + 2m - 2$
        
        \textbf{Step 7:}
                $D(m) = 2k + 2m - 2$
        
        \textbf{Step 8:}
                $D(m) = 2(k + m - 1)$
\end{enumerate}
